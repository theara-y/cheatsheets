SQL Cheat Sheet

Beginner (Essentials)

Basic Queries

SELECT

(Basic selection)

SELECT 
  column1, column2 
  FROM table_name;

WHERE

(Filtering records)

SELECT * 
  FROM table_name 
  WHERE column = value;

Sorting and Limiting

ORDER BY

(Sorting results)

SELECT * 
  FROM table_name 
  ORDER BY column ASC|DESC;

LIMIT/OFFSET

(Limiting results)

SELECT * 
  FROM table_name 
  LIMIT 5 
  OFFSET 10;

Aggregation Functions

COUNT(), SUM(), AVG(), MAX(), MIN()

SELECT COUNT(*) 
  FROM table_name;

SELECT AVG(column) 
  FROM table_name;

INSERT

(Adding data)

INSERT INTO 
  table_name 
    (column1, column2) 
  VALUES 
    (value1, value2);

UPDATE

(Modifying data)

UPDATE table_name 
  SET column1 = value1 
  WHERE condition;

DELETE

(Removing data)

DELETE 
  FROM table_name 
  WHERE condition;

Intermediate (Core Concepts)

JOIN Operations

Combining Tables

INNER JOIN

SELECT columns 
  FROM table1 INNER 
  JOIN table2 
  ON table1.id = table2.id;

LEFT JOIN

SELECT columns 
  FROM table1 
  LEFT JOIN table2 
  ON table1.id = table2.id;

RIGHT JOIN

SELECT columns 
  FROM table1 
  RIGHT JOIN table2 
  ON table1.id = table2.id;

FULL OUTER JOIN

SELECT columns 
  FROM table1 
  FULL OUTER JOIN table2 
  ON table1.id = table2.id;

Subqueries (Nested Queries)

Subquery in SELECT

SELECT 
  column, 
  (SELECT COUNT(*) 
    FROM table2 
    WHERE 
    table2.column = table1.column
  ) AS count 
  FROM table1;

Subquery in WHERE

SELECT column 
  FROM table 
  WHERE column IN (
    SELECT column 
      FROM table2 
      WHERE condition
  );

Group By / Having (Grouping & Filtering)

GROUP BY

(Aggregating by category)

SELECT column, COUNT(*) 
  FROM table_name 
  GROUP BY column;

HAVING

(Filtering after GROUP BY)

SELECT column, COUNT(*) 
  FROM table_name 
  GROUP BY column 
  HAVING COUNT(*) > 10;

Removing Duplicates

DISTINCT (Removing Duplicates)

SELECT DISTINCT column 
  FROM table_name;

Indexing (Performance)

Indexing (Basic indexing for performance)

CREATE 
  INDEX index_name 
  ON table_name (column);

Advanced (Real-World Use)

CTEs (Common Table Expression)

Basic CTE

WITH cte AS (
  SELECT column1, column2 
    FROM table_name 
    WHERE condition
)
SELECT * FROM cte;

Recursive CTE

WITH RECURSIVE cte AS (
  SELECT id, parent_id 
    FROM table_name 
    WHERE parent_id IS NULL
  UNION ALL
  SELECT t.id, t.parent_id 
    FROM table_name t 
    INNER JOIN cte c 
    ON c.id = t.parent_id
)
SELECT * FROM cte;

Window Functions (Advanced aggregation over a subset of rows)

ROW_NUMBER(), RANK(), DENSE_RANK()

SELECT 
  column1, 
  ROW_NUMBER() OVER (
    PARTITION BY column2 
    ORDER BY column3
  ) 
  FROM table_name;

SUM(), AVG() OVER (Windowing Functions)

SELECT 
  column1, 
  SUM(column2) OVER (
    PARTITION BY column3
  ) 
  FROM table_name;

Transactions

BEGIN TRANSACTION;
COMMIT;
ROLLBACK;

Stored Procedures & Functions

Stored Procedure

CREATE 
  PROCEDURE procedure_name 
AS
BEGIN
  SELECT * FROM table_name;
END;

User-Defined Function (UDF)

CREATE 
  FUNCTION 
    function_name (@param datatype) 
  RETURNS 
    datatype
AS
BEGIN
  RETURN (
    SELECT column 
    FROM table_name 
    WHERE column = @param
  );
END;

Triggers

(Automating action based on changes)

CREATE 
  TRIGGER trigger_name
  AFTER INSERT 
  ON table_name
  FOR EACH ROW
BEGIN
-- Action code here
END;

Expert (Optimization & Problem Solving)

Query Optimization Techniques

EXPLAIN PLAN

(Viewing query execution plan)

EXPLAIN SELECT * FROM table_name;

Index Optimization PLAN

(Choosing indexes for faster retrieval)

  • Use composite indexes
  • Indexing foreign keys
  • Avoid over-indexing (too many indexes can slow down writes)

Joins Optimization

(Minimize the number of rows involved)

  • Choose appropriate join types

Partitioning & Sharding

Dividing large tables into smaller parts

CREATE 
  TABLE table_name 
PARTITION 
  BY RANGE (date_column);

Handling Large Datasets

BULK INSERT / COPY

(For fast data loading)

BULK INSERT table_name 
  FROM 'file_path';

Window Functions for Large Aggregations

  • Perform calculations over partitions without collapsing the data.

Master (Design & Advanced Problem-Solving)

Data Modeling & Normalization

(Database Design)

  • 1NF, 2NF, 3NF
  • Star Schema / Snowflake Schema for Data Warehousing

Advanced Query Design for Real-World Scenarios

  • Recursive Queries
  • Pivoting & Unpivoting Data

Pivoting & Unpivoting Data

SELECT * 
FROM table_name 
PIVOT (
  SUM(column) 
  FOR column2 
  IN ([value1], [value2])
) AS pvt;

Multi-Database Design

(For cross-database queries or microservices)

Federated Queries

(Accessing different databases in a single query)

SELECT * 
  FROM database1.table_name 
UNION 
  SELECT * 
FROM database2.table_name;

Advanced Data Security:

  • Row-Level Security

    (Restrict access to certain rows based on user)

  • Encryption at Rest and in Transit

Replication & Backup Strategies:

  • Master-Slave Replication for high availability
  • Point-in-Time Recovery for backup and disaster recovery
